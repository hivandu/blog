```
public static void reverse(int x[]){
    int foot=0;
    int head=0;
    if(x.length%2==0){
        foot=x.length/2;
            head=foot-1;
            for(int i=0; i int temp=x[head];
            x[head]=x[foot];
            x[foot]=temp;
            head--;
            foot++;
        }
    }else{
        foot=x.length/2;
        head=foot;
        for(int i=0; i int temp=x[head];
        x[head]=x[foot];
        x[foot]=temp;
        head--;
        foot++;
        }
    }
}
```
